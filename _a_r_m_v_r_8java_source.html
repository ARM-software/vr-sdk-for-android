65  private void extractAsset(String assetName)
66  {
67  File file = new File(assetsDirectory + assetName);
68 
69  if(file.exists()) {
70  Log.d(LOGTAG, assetName + " already exists. No extraction needed.\n");
71  } else {
72  Log.d(LOGTAG, assetName + " doesn't exist. Extraction needed. \n");
73 
74  try {
75  RandomAccessFile randomAccessFile = new RandomAccessFile(assetsDirectory + assetName,"rw");
76  AssetManager assetManager = applicationContext.getResources().getAssets();
77  InputStream inputStream = assetManager.open(assetName);
78 
79  byte buffer[] = new byte[1024];
80  int count = inputStream.read(buffer, 0, 1024);
81 
82  while (count > 0) {
83  randomAccessFile.write(buffer, 0, count);
84 
85  count = inputStream.read(buffer, 0, 1024);
86  }
87 
88  randomAccessFile.close();
89  inputStream.close();
90  } catch(Exception e) {
91  Log.e(LOGTAG, "Failure in extractAssets(): " + e.toString() + " " + assetsDirectory + assetName);
92  }
93 
94  if(file.exists()) {
95  Log.d(LOGTAG,"File extracted successfully");
96  }
97  }
98  }
